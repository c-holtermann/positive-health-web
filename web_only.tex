\documentclass[varwidth=80cm]{standalone}
\usepackage{tikz}
\usetikzlibrary{shapes}

\begin{document}

\newcommand{\D}{6} % number of dimensions (config option)
\newcommand{\U}{10} % number of scale units (config option)

\newdimen\R % maximal diagram radius (config option)
\R=3.5cm 
\newdimen\L % radius to put dimension labels (config option)
\L=4.4cm

\newcommand{\A}{360/\D} % calculated angle between dimension axes
\newcommand{\anglebase}{\A/2}
\newcommand{\AX}{\X*\A+\anglebase} % calculate angles so that one spike shows down

\begin{figure}[htbp]
 \centering

\begin{tikzpicture}[scale=1]
  \tikzstyle{label} = [align=center, text width=5em, font=\small]
  \path (0:0cm) coordinate (O); % define coordinate for origin

  % draw the spiderweb
  \foreach \X in {1,...,\D}{
    \draw (\AX:0) -- (\AX:\R);
  }

  \foreach \Y in {0,...,\U}{
    \foreach \X in {1,...,\D}{
      \path (\AX:\Y*\R/\U) coordinate (D\X-\Y);
      \fill (D\X-\Y) circle (1pt);
    }
    \draw [opacity=0.3] (\anglebase:\Y*\R/\U) \foreach \X in {1,...,\D}{
        -- (\AX:\Y*\R/\U)
    } -- cycle;
  }

  % define labels for each dimension axis (names config option)
  \path (1*\A+\anglebase:\L) node[label] (L1) {Bodily\\functions};
  \path (2*\A+\anglebase:\L) node[label] (L2) {Mental functions \&\\perceptions};
  \path (3*\A+\anglebase:\L) node[label] (L3) {Spiritual\\dimension};
  \path (4*\A+\anglebase:\L) node[label] (L4) {Quality\\of life};
  \path (5*\A+\anglebase:\L) node[label] (L5) {Social \& Societal\\participation};
  \path (6*\A+\anglebase:\L) node[label] (L6) {Daily\\functioning};

\end{tikzpicture}
\end{figure}

\end{document}
